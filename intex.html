from PIL import Image, ImageDraw, ImageFont width, height = 400, 400 image = Image.new('RGB', (width, height), color='white') draw = ImageDraw.Draw(image) draw.ellipse((100, 100, 300, 300), fill='blue', outline='black' font = ImageFont.load_default() text = "This is a bird!" draw.text((120, 320), text, font=font, fill='black') image.save('bird_image.png')
